<htpl>
    <main>
        <in val,val1,res/>
        <take val,val1/>
        <? if(val>val1)
            <fx res=gcd_algo(val1,val) />
            <log>${res}</log>
        ?>
        <? elif(val<val1)
            <fx res=gcd_algo(val,val1) />
            <log>${res}\n</log>
        ?>
        <? else
            <log>${val}\n</log>
        ?>
    </main>
    <fx in gcd_algo(in a,in b)>
        <? if(a==0)
            <throw b/> 
        ?>
        <? else
            <throw gcd_algo(b%a,a)/> 
        ?>
    </gcd_algo>
</htpl>
